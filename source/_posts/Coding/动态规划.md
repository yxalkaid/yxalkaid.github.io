### 背包DP
- 0-1背包
    ```Python
    def knapsack(capacity, weights, values):
        dp = [0] * (capacity + 1)
        for i in range(len(weights)):
            weight = weights[i]
            value = values[i]
            # 倒序遍历确保每个物品只选一次
            for j in range(capacity, weight - 1, -1):
                dp[j] = max(dp[j], dp[j - weight] + value)
        return dp[-1]
    ```

- 多重背包
    ```Python
    def knapsack_multiple(capacity, weights, values, counts):
        dp = [0] * (capacity + 1)
        for i in range(len(weights)):
            weight = weights[i]
            value = values[i]
            count = counts[i]
            for j in range(capacity, weight - 1, -1):
                for k in range(1, count):
                    if weight * k > j:
                        break
                    dp[j] = max(dp[j], dp[j - weight * k] + value * k)
        return dp[-1]
    ```

    ```Python
    def knapsack_multiple(capacity, weights, values, counts):
        dp = [0] * (capacity + 1)
        for i in range(len(weights)):
            weight = weights[i]
            value = values[i]
            count = counts[i]
            k = 1
            while k < count:
                # 二进制拆分：将当前k个物品视为01背包的子物品
                for j in range(capacity, weight * k - 1, -1):
                    dp[j] = max(dp[j], dp[j - weight * k] + value * k)
                count -= k
                k *= 2
            # 处理剩余无法拆分的cnt个物品
            if count > 0:
                for j in range(capacity, weight * count - 1, -1):
                    dp[j] = max(dp[j], dp[j - weight * count] + value * count)
        return dp[-1]
    ```

- 完全背包
    ```Python
    def knapsack_complete(capacity, weights, values):
        dp = [0] * (capacity + 1)
        for i in range(len(weights)):
            weight = weights[i]
            value = values[i]
            
            # 正序遍历允许重复选择当前物品
            for j in range(weight, capacity + 1):
                dp[j] = max(dp[j], dp[j - weight] + value)
    ```
